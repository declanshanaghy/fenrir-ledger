          <div className="work-body">
            <p><span className="mono">crypto.randomUUID()</span> is restricted to secure contexts (HTTPS / localhost). Dev servers accessed over a LAN IP — common on local network builds — silently lack it.</p>
            <p>Added <span className="hl">generateId()</span> to <span className="mono">card-utils.ts</span>: uses <span className="mono">randomUUID</span> when available, falls back to <span className="mono">getRandomValues</span> which is unrestricted and produces a valid UUID v4.</p>
          </div>

          <div className="code-block">
            <pre><span className="ca">export function generateId(): string {</span>
<span className="ca">  if (typeof crypto.randomUUID === "function") {</span>
<span className="ca">    return crypto.randomUUID();           // secure context</span>
<span className="ca">  }</span>
<span className="cc">  // Fallback — works over HTTP / LAN IP</span>
<span className="ca">  const bytes = crypto.getRandomValues(new Uint8Array(16));</span>
<span className="ca">  bytes[6] = (bytes[6]! & 0x0f) | 0x40;  // version 4</span>
<span className="ca">  bytes[8] = (bytes[8]! & 0x3f) | 0x80;  // variant bits</span>
<span className="ca">  const hex = Array.from(bytes).map(b => b.toString(16).padStart(2, "0"));</span>
<span className="ca">  return `${hex.slice(0,4).join("")}-...-${hex.slice(10).join("")}`;</span>
<span className="ca">}</span></pre>
          </div>

        <div className="work-card">
          <div className="work-body">
            <p>Added <span className="mono">scrollToFirstError</span> as the second argument to <span className="mono">handleSubmit</span>. Candidates are sorted by <span className="mono">compareDocumentPosition</span> — DOM order, not schema insertion order — so the scroll always targets the error that appears <em>first on screen</em>.</p>
            <p>Set <span className="mono">shouldFocusError: false</span> in <span className="mono">useForm</span> to prevent react-hook-form's built-in focus from fighting the manual scroll.</p>
          </div>

          <div className="code-block">
            <pre><span className="cc">// useForm — disable built-in focus</span>
<span className="ca">shouldFocusError: false,</span>

<span className="cc">// Scroll handler — passed as handleSubmit second arg</span>
<span className="ca">const scrollToFirstError = (errs: Record&lt;string, unknown&gt;) => {</span>
<span className="ca">  const elements = Object.keys(errs)</span>
<span className="ca">    .map(key => document.getElementById(key))</span>
<span className="ca">    .filter((el): el is HTMLElement => el !== null)</span>
<span className="ca">    .sort((a, b) =></span>
<span className="ca">      a.compareDocumentPosition(b) & Node.DOCUMENT_POSITION_FOLLOWING ? -1 : 1</span>
<span className="ca">    );</span>
<span className="ca">  if (elements.length > 0) {</span>
<span className="ca">    elements[0]!.scrollIntoView({ behavior: "smooth", block: "center" });</span>
<span className="ca">    elements[0]!.focus();</span>
<span className="ca">  }</span>
<span className="ca">};</span>

<span className="cc">// Form tag</span>
<span className="ca">&lt;form onSubmit={handleSubmit(onSubmit, scrollToFirstError)}&gt;</span></pre>
          </div>
